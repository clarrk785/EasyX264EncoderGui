				        private string ffmpegArg(string input)
        {
            return string.Format(" -i \"{0}\" -f wav ",
                input);
        }

        private string neroaacArg(string output)
        {
            return string.Format(" -ignorelength -q {2} -lc -if - -of \"{1}\" ",
                output);
